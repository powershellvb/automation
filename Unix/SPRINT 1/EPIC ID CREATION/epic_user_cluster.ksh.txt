#!/bin/ksh
sudo su -

host=`uname -n`
IFS=:
while read userid gecos;do
userverify=`cat /etc/passwd | grep -i $userid | wc -l`
if [[ $userverify = 0 ]]
then
$rg=`clRGinfo -v | grep -i "Resource Group" | awk -F ":" '{print $2}'`
/usr/es/sbin/cluster/cspoc/fix_args nop cl_mkuser -cspoc-g '$rg' pgrp='epicuser' groups='epicuser' home='/home/$userid' gecos='$gecos' SYSTEM='KRB5files' registry='KRB5files' $userid
if [[ $? = 0 ]]
then
echo "User $userid created sucessfully"
else
echo "User account creation failed for the ID $userid in $host"
exit
exit
fi
else
echo "User exist in $host"
exit
exit
fi
done</tmp/input_epic.txt
sudo -u epicadm /epic/custom/bin/cacheadd.ksh $userid
exit
exit
